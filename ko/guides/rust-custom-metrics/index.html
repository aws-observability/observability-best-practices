AWS Rust SDKлЎң м»ӨмҠӨн…Җ Metrics мғқм„ұн•ҳкё°

мҶҢк°ңвҖӢ

RustлҠ” м•Ҳм „м„ұ, м„ұлҠҘ, лҸҷмӢңм„ұм—җ мӨ‘м җмқ„ л‘” мӢңмҠӨн…ң н”„лЎңк·ёлһҳл°Қ м–ём–ҙлЎң, мҶҢн”„нҠёмӣЁм–ҙ к°ңл°ң л¶„м•јм—җм„ң кҫёмӨҖнһҲ мқёкё°лҘј м–»кі  мһҲмҠөлӢҲлӢӨ. л©”лӘЁлҰ¬ кҙҖлҰ¬мҷҖ мҠӨл Ҳл“ң м•Ҳм „м„ұм—җ лҢҖн•ң лҸ…нҠ№н•ң м ‘к·ј л°©мӢқ лҚ•л¶„м—җ, м•Ҳм •м Ғмқҙкі  нҡЁмңЁм Ғмқё м• н”ҢлҰ¬мјҖмқҙм…ҳмқ„ кө¬м¶•н•ҳлҠ” лҚ° л§Өл Ҙм Ғмқё м„ нғқм§Җк°Җ лҗҳкі  мһҲмңјл©°, нҠ№нһҲ нҒҙлқјмҡ°л“ң нҷҳкІҪм—җм„ң к·ё к°•м җмқҙ л‘җл“ңлҹ¬м§‘лӢҲлӢӨ. м„ңлІ„лҰ¬мҠӨ м•„нӮӨн…ҚмІҳк°Җ нҷ•мӮ°лҗҳкі  кі м„ұлҠҘВ·нҷ•мһҘ к°ҖлҠҘн•ң м„ңл№„мҠӨк°Җ н•„мҡ”н•ҙм§Җл©ҙм„ң, Rustмқҳ м—ӯлҹүмқҖ нҒҙлқјмҡ°л“ң л„ӨмқҙнӢ°лёҢ м• н”ҢлҰ¬мјҖмқҙм…ҳ кө¬м¶•м—җ нғҒмӣ”н•ң м„ нғқмқҙ лҗҳм—ҲмҠөлӢҲлӢӨ. мқҙ к°Җмқҙл“ңм—җм„ңлҠ” AWS Rust SDKлҘј нҷңмҡ©н•ҳм—¬ CloudWatch м»ӨмҠӨн…Җ metricsлҘј мғқм„ұн•ҳлҠ” л°©лІ•мқ„ мӮҙнҺҙлҙ…лӢҲлӢӨ. мқҙлҘј нҶөн•ҙ AWS м—җмҪ”мӢңмҠӨн…ң лӮҙм—җм„ң м• н”ҢлҰ¬мјҖмқҙм…ҳмқҳ м„ұлҠҘкіј лҸҷмһ‘м—җ лҢҖн•ң лҚ” к№ҠмқҖ мқёмӮ¬мқҙнҠёлҘј нҷ•ліҙн•  мҲҳ мһҲмҠөлӢҲлӢӨ.

мӮ¬м „ мӨҖл№„ мӮ¬н•ӯвҖӢ

мқҙ к°Җмқҙл“ңлҘј л”°лқјн•ҳл Өл©ҙ RustлҘј м„Өм№ҳн•ҳкі , мқҙнӣ„ мӮ¬мҡ©н•  лҚ°мқҙн„°лҘј м ҖмһҘн•  CloudWatch log groupкіј log streamмқ„ мғқм„ұн•ҙм•ј н•©лӢҲлӢӨ.

Rust м„Өм№ҳвҖӢ

Mac лҳҗлҠ” Linux нҷҳкІҪ:

curl --proto '=https' --tlsv1.2 -sSf https://sh.rustup.rs | sh

Windows нҷҳкІҪм—җм„ңлҠ” rustup-init.exeлҘј лӢӨмҡҙлЎңл“ңн•ҳм—¬ мӢӨн–үн•©лӢҲлӢӨ.

CloudWatch Log Group л°Ҹ Log Stream мғқм„ұвҖӢ

  1. CloudWatch Log Group мғқм„ұ:
aws logs create-log-group --log-group-name rust_custom
  1. CloudWatch Log Stream мғқм„ұ:
aws logs create-log-stream --log-group-name rust_custom --log-stream-name diceroll_log_stream

мҪ”л“ңвҖӢ

м „мІҙ мҪ”л“ңлҠ” мқҙ м ҖмһҘмҶҢмқҳ sandbox м„№м…ҳм—җм„ң нҷ•мқён•  мҲҳ мһҲмҠөлӢҲлӢӨ.

git clone https://github.com/aws-observability/observability-best-practices.git
cd observability-best-practices/sandbox/rust-custom-metrics

мқҙ мҪ”л“ңлҠ” лЁјм Җ мЈјмӮ¬мң„ көҙлҰ¬кё°лҘј мӢңл®¬л Ҳмқҙм…ҳн•©лӢҲлӢӨ. мЈјмӮ¬мң„ к°’мқ„ м»ӨмҠӨн…Җ metricмңјлЎң мӮ¬мҡ©н•ңлӢӨкі  к°Җм •н•ҳкІ мҠөлӢҲлӢӨ. к·ёлҹ° лӢӨмқҢ мқҙ metricмқ„ CloudWatchм—җ м¶”к°Җн•ҳкі  лҢҖмӢңліҙл“ңм—җм„ң нҷ•мқён•ҳлҠ” 3к°Җм§Җ л°©лІ•мқ„ ліҙм—¬л“ңлҰ¬кІ мҠөлӢҲлӢӨ.

м• н”ҢлҰ¬мјҖмқҙм…ҳ м„Өм •вҖӢ

лЁјм Җ м• н”ҢлҰ¬мјҖмқҙм…ҳм—җм„ң мӮ¬мҡ©н•  crateл“Өмқ„ importн•ҙм•ј н•©лӢҲлӢӨ.

use crate::cloudwatch::types::Dimension;
use crate::cloudwatchlogs::types::InputLogEvent;
use aws_sdk_cloudwatch as cloudwatch;
use aws_sdk_cloudwatch::config::BehaviorVersion;
use aws_sdk_cloudwatch::types::MetricDatum;
use aws_sdk_cloudwatchlogs as cloudwatchlogs;
use rand::prelude::*;
use serde::Serialize;
use serde_json::json;
use std::time::{SystemTime, UNIX_EPOCH};

мқҙ import лё”лЎқм—җм„ңлҠ” мЈјлЎң мӮ¬мҡ©н•  AWS SDK лқјмқҙлёҢлҹ¬лҰ¬лҘј к°Җм ёмҳөлӢҲлӢӨ. лһңлҚӨ мЈјмӮ¬мң„ к°’мқ„ мғқм„ұн•ҳкё° мң„н•ҙ 'rand' crateлҸ„ нҸ¬н•Ён•©лӢҲлӢӨ. л§Ҳм§Җл§үмңјлЎң 'serde'мҷҖ 'time' к°ҷмқҖ лқјмқҙлёҢлҹ¬лҰ¬лҘј мӮ¬мҡ©н•ҳм—¬ SDK нҳём¶ңм—җ н•„мҡ”н•ң лҚ°мқҙн„°лҘј мғқм„ұн•©лӢҲлӢӨ.

мқҙм ң main н•ЁмҲҳм—җм„ң мЈјмӮ¬мң„ к°’мқ„ мғқм„ұн•  мҲҳ мһҲмҠөлӢҲлӢӨ. мқҙ к°’мқҖ мқҙнӣ„ 3к°Җм§Җ AWS SDK нҳём¶ң лӘЁл‘җм—җм„ң мӮ¬мҡ©лҗ©лӢҲлӢӨ.

//select a random number 1-6 to represent a diceroll
let mut rng = rand::thread_rng();
let roll_value = rng.gen_range(1..7);

мЈјмӮ¬мң„ к°’мқ„ м–»м—ҲмңјлҜҖлЎң, мқҙм ң мқҙ к°’мқ„ CloudWatch м»ӨмҠӨн…Җ metricмңјлЎң м¶”к°Җн•ҳлҠ” 3к°Җм§Җ л°©лІ•мқ„ мӮҙнҺҙліҙкІ мҠөлӢҲлӢӨ. к°’мқҙ м»ӨмҠӨн…Җ metricмңјлЎң л“ұлЎқлҗҳл©ҙ, н•ҙлӢ№ к°’м—җ лҢҖн•ҙ м•ҢлһҢ м„Өм •, мқҙмғҒ нғҗм§Җ кө¬м„ұ, лҢҖмӢңліҙл“ңм—җ мӢңк°Ғнҷ”н•ҳлҠ” л“ұ лӢӨм–‘н•ң кё°лҠҘмқ„ нҷңмҡ©н•  мҲҳ мһҲмҠөлӢҲлӢӨ.

Put Metric DataвҖӢ

CloudWatchм—җ к°’мқ„ м¶”к°Җн•ҳлҠ” мІ« лІҲм§ё л°©лІ•мқҖ PutMetricDataмһ…лӢҲлӢӨ. PutMetricDataлҘј мӮ¬мҡ©н•ҳл©ҙ metricмқҳ мӢңкі„м—ҙ к°’мқ„ CloudWatchм—җ м§Ғм ‘ кё°лЎқн•©лӢҲлӢӨ. мқҙкІғмқҙ к°’мқ„ м¶”к°Җн•ҳлҠ” к°ҖмһҘ нҡЁмңЁм Ғмқё л°©лІ•мһ…лӢҲлӢӨ. PutMetricDataлҘј мӮ¬мҡ©н•  л•ҢлҠ” metric к°’кіј н•Ёк»ҳ namespace, к·ёлҰ¬кі  к°Ғ AWS SDK нҳём¶ңм—җ dimensionмқ„ н•Ёк»ҳ м ңкіөн•ҙм•ј н•©лӢҲлӢӨ. мҪ”л“ңлҘј мӮҙнҺҙліҙкІ мҠөлӢҲлӢӨ:

лЁјм Җ metric(мЈјмӮ¬мң„ к°’)мқ„ л°ӣм•„ Result нғҖмһ…мқ„ л°ҳнҷҳн•ҳлҠ” н•ЁмҲҳлҘј м„Өм •н•©лӢҲлӢӨ. Rustм—җм„ң ResultлҠ” м„ұкіө лҳҗлҠ” мӢӨнҢЁлҘј лӮҳнғҖлғ…лӢҲлӢӨ. н•ЁмҲҳ лӮҙм—җм„ң к°ҖмһҘ лЁјм Җ н•ҳлҠ” мқјмқҖ AWS Rust SDK нҒҙлқјмқҙм–ёнҠёлҘј мҙҲкё°нҷ”н•ҳлҠ” кІғмһ…лӢҲлӢӨ. нҒҙлқјмқҙм–ёнҠёлҠ” лЎңм»¬ нҷҳкІҪм—җм„ң мһҗкІ© мҰқлӘ…кіј лҰ¬м „мқ„ мғҒмҶҚл°ӣмҠөлӢҲлӢӨ. л”°лқјм„ң мқҙ мҪ”л“ңлҘј мӢӨн–үн•ҳкё° м „м—җ м»Өл§Ёл“ң лқјмқём—җм„ң aws configureлҘј мӢӨн–үн•ҳм—¬ м„Өм •мқҙ мҷ„лЈҢлҗҳм–ҙ мһҲлҠ”м§Җ нҷ•мқён•ҳм„ёмҡ”.

async fn put_metric_data(roll_value: i32) -> Result<(), cloudwatch::Error> {
    //Create a reusable aws config that we can pass to our clients
    let config = aws_config::load_defaults(BehaviorVersion::v2023_11_09()).await;

    //Create a cloudwatch client
    let client = cloudwatch::Client::new(&config);

нҒҙлқјмқҙм–ёнҠёлҘј мҙҲкё°нҷ”н•ң нӣ„, PutMetricData API нҳём¶ңм—җ н•„мҡ”н•ң мһ…л Ҙк°’мқ„ м„Өм •н•  мҲҳ мһҲмҠөлӢҲлӢӨ. dimensionмқ„ м •мқҳн•ң лӢӨмқҢ, dimensionкіј к°’мқ„ мЎ°н•©н•ң MetricDatum мһҗмІҙлҘј м •мқҳн•ҙм•ј н•©лӢҲлӢӨ.

//Use fluent builders to build the required input for pmd call, starting with dimensions.
let dimensions = Dimension::builder()
    .name("roll_value_pmd_dimension")
    .value(roll_value.to_string())
    .build();

let put_metric_data_input = MetricDatum::builder()
    .metric_name("roll_value_pmd")
    .dimensions(dimensions)
    .value(f64::from(roll_value))
    .build();

л§Ҳм§Җл§үмңјлЎң м•һм„ң м •мқҳн•ң мһ…л Ҙк°’мқ„ мӮ¬мҡ©н•ҳм—¬ PutMetricData API нҳём¶ңмқ„ мҲҳн–үн•©лӢҲлӢӨ.

let response = client
    .put_metric_data()
    .namespace("rust_custom_metrics")
    .metric_data(put_metric_data_input)
    .send()
    .await?;
println!("Metric Submitted: {:?}", response);
Ok(())

SDK нҳём¶ңмқҙ async н•ЁмҲҳ лӮҙм—җ мһҲлӢӨлҠ” м җм—җ мЈјлӘ©н•ҳм„ёмҡ”. н•ЁмҲҳк°Җ л№„лҸҷкё°м ҒмңјлЎң мҷ„лЈҢлҗҳлҜҖлЎң, мҷ„лЈҢлҗ  л•Ңк№Ңм§Җ awaitн•ҙм•ј н•©лӢҲлӢӨ. к·ёлҹ° лӢӨмқҢ н•ЁмҲҳ мөңмғҒмң„м—җ м •мқҳн•ң лҢҖлЎң Result нғҖмһ…мқ„ л°ҳнҷҳн•©лӢҲлӢӨ.

mainм—җм„ң н•ЁмҲҳлҘј нҳём¶ңн•  л•ҢлҠ” лӢӨмқҢкіј к°ҷмқҙ мһ‘м„ұн•©лӢҲлӢӨ:

//call the put_metric_data function with the roll value
println!("First we will write a custom metric with PutMetricData API call");
put_metric_data(roll_value).await.unwrap();

м—¬кё°м„ңлҸ„ н•ЁмҲҳ нҳём¶ңмқҙ мҷ„лЈҢлҗ  л•Ңк№Ңм§Җ awaitн•ң лӢӨмқҢ, к°’мқ„ unwrapн•©лӢҲлӢӨ. мқҙ мҳҲм ңм—җм„ңлҠ” 'Ok' кІ°кіјл§Ң кҙҖмӢ¬мқҙ мһҲкі  м—җлҹ¬лҠ” мІҳлҰ¬н•ҳм§Җ м•Ҡкё° л•Ңл¬ёмһ…лӢҲлӢӨ. н”„лЎңлҚ•м…ҳ нҷҳкІҪм—җм„ңлҠ” лӢӨлҘё л°©мӢқмңјлЎң м—җлҹ¬лҘј мІҳлҰ¬н•ҳлҠ” кІғмқҙ мқјл°ҳм Ғмһ…лӢҲлӢӨ.

PutLogEvent + Metric FilterвҖӢ

м»ӨмҠӨн…Җ metricмқ„ мғқм„ұн•ҳлҠ” л‘җ лІҲм§ё л°©лІ•мқҖ к°’мқ„ CloudWatch log groupм—җ кё°лЎқн•ҳлҠ” кІғмһ…лӢҲлӢӨ. metricмқҙ CloudWatch log groupм—җ м ҖмһҘлҗҳл©ҙ, Metric FilterлҘј мӮ¬мҡ©н•ҳм—¬ лЎңк·ё лҚ°мқҙн„°м—җм„ң metric лҚ°мқҙн„°лҘј м¶”м¶ңн•  мҲҳ мһҲмҠөлӢҲлӢӨ.

лЁјм Җ лЎңк·ё л©”мӢңм§ҖлҘј мң„н•ң structлҘј м •мқҳн•©лӢҲлӢӨ. JSONмқ„ мҲҳлҸҷмңјлЎң л§Ңл“Ө мҲҳлҸ„ мһҲмңјлҜҖлЎң мқҙ лӢЁкі„лҠ” м„ нғқ мӮ¬н•ӯмқҙм§Җл§Ң, лҚ” ліөмһЎн•ң м• н”ҢлҰ¬мјҖмқҙм…ҳм—җм„ңлҠ” мһ¬мӮ¬мҡ©м„ұмқ„ мң„н•ҙ мқҙлҹ¬н•ң лЎңк№… structлҘј мӮ¬мҡ©н•ҳлҠ” кІғмқҙ мўӢмҠөлӢҲлӢӨ.

//Make a simple struct for the log message. We could also just create a json string manually.
#[derive(Serialize)]
struct DicerollValue {
    welcome_message: String,
    roll_value: i32,
}

structк°Җ м •мқҳлҗҳл©ҙ AWS API нҳём¶ңмқ„ н•  мӨҖл№„к°Җ лҗ©лӢҲлӢӨ. мқҙлІҲм—җлҠ” logs SDKлҘј мӮ¬мҡ©н•ҳм—¬ API нҒҙлқјмқҙм–ёнҠёлҘј мғқм„ұн•©лӢҲлӢӨ. лҳҗн•ң unix epoch нғҖмқҙл°ҚмңјлЎң мӢңмҠӨн…ң мӢңк°„мқ„ м •мқҳн•©лӢҲлӢӨ.

//Create a reusable aws config that we can pass to our clients
let config = aws_config::load_defaults(BehaviorVersion::v2023_11_09()).await;

//Create a cloudwatch logs client
let client = cloudwatchlogs::Client::new(&config);

//Let's get the time in ms from unix epoch, this is required for CWlogs
let time_now = SystemTime::now()
    .duration_since(UNIX_EPOCH)
    .unwrap()
    .as_millis() as i64;

лЁјм Җ м•һм„ң м •мқҳн•ң structмқҳ мғҲ мқёмҠӨн„ҙмҠӨлЎңл¶Җн„° JSONмқ„ мғқм„ұн•©лӢҲлӢӨ. к·ёлҹ° лӢӨмқҢ мқҙлҘј мӮ¬мҡ©н•ҳм—¬ log eventлҘј л§Ңл“ӯлӢҲлӢӨ.

let log_json = json!(DicerollValue {
    welcome_message: String::from("Hello from rust!"),
    roll_value
});

let log_event = InputLogEvent::builder()
    .timestamp(time_now)
    .message(log_json.to_string())
    .build();

мқҙм ң PutMetricDataмҷҖ мң мӮ¬н•ң л°©мӢқмңјлЎң API нҳём¶ңмқ„ мҷ„лЈҢн•  мҲҳ мһҲмҠөлӢҲлӢӨ.

let response = client
    .put_log_events()
    .log_group_name("rust_custom")
    .log_stream_name("diceroll_log_stream")
    .log_events(log_event.unwrap())
    .send()
    .await?;

println!("Log event submitted: {:?}", response);
Ok(())

log eventк°Җ м ңм¶ңлҗҳл©ҙ, CloudWatchм—җм„ң н•ҙлӢ№ log groupм—җ лҢҖн•ң Metric FilterлҘј мғқм„ұн•ҳм—¬ metricмқ„ мҳ¬л°”лҘҙкІҢ м¶”м¶ңн•ҙм•ј н•©лӢҲлӢӨ.

CloudWatch мҪҳмҶ”м—җм„ң мғқм„ұн•ң rust_custom log groupмңјлЎң мқҙлҸҷн•©лӢҲлӢӨ. к·ёлҹ° лӢӨмқҢ metric filterлҘј мғқм„ұн•©лӢҲлӢӨ. н•„н„° нҢЁн„ҙмқҖ {$.roll_value = *}лЎң м„Өм •н•©лӢҲлӢӨ. Metric Valueм—җлҠ” $.roll_valueлҘј мӮ¬мҡ©н•©лӢҲлӢӨ. namespaceмҷҖ metric мқҙлҰ„мқҖ мӣҗн•ҳлҠ” лҢҖлЎң м„Өм •н•  мҲҳ мһҲмҠөлӢҲлӢӨ. мқҙ Metric FilterлҠ” лӢӨмқҢкіј к°ҷмқҙ м„ӨлӘ…н•  мҲҳ мһҲмҠөлӢҲлӢӨ:

"'roll_value'лқјлҠ” н•„л“ңк°Җ л“Өм–ҙмҳ¬ л•Ңл§ҲлӢӨ, к°’м—җ кҙҖкі„м—Ҷмқҙ н•„н„°лҘј нҠёлҰ¬кұ°н•©лӢҲлӢӨ. нҠёлҰ¬кұ°лҗҳл©ҙ 'roll_value'лҘј CloudWatch Metricsм—җ кё°лЎқн•  мҲ«мһҗлЎң мӮ¬мҡ©н•©лӢҲлӢӨ."

мқҙ л°©лІ•мқҖ лЎңк·ё нҳ•мӢқмқ„ м§Ғм ‘ м ңм–ҙн•  мҲҳ м—ҶлҠ” мғҒнҷ©м—җм„ң лЎңк·ё лҚ°мқҙн„°лЎңл¶Җн„° мӢңкі„м—ҙ к°’мқ„ м¶”м¶ңн•  л•Ң л§Өмҡ° мң мҡ©н•©лӢҲлӢӨ. м§ҖкёҲмІҳлҹј мҪ”л“ңлҘј м§Ғм ‘ кі„мёЎн•ҳлҠ” кІҪмҡ°м—җлҠ” лЎңк·ё лҚ°мқҙн„° нҳ•мӢқмқ„ м ңм–ҙн•  мҲҳ мһҲмңјлҜҖлЎң, лӢӨмқҢ лӢЁкі„м—җм„ң м„ӨлӘ…н•  CloudWatch Embedded Metric Formatмқ„ мӮ¬мҡ©н•ҳлҠ” кІғмқҙ лҚ” лӮҳмқҖ л°©лІ•мқј мҲҳ мһҲмҠөлӢҲлӢӨ.

PutLogEvent + Embedded Metric FormatвҖӢ

CloudWatch Embedded Metric Format(EMF)мқҖ мӢңкі„м—ҙ metricsлҘј лЎңк·ём—җ м§Ғм ‘ мһ„лІ л“ңн•ҳлҠ” л°©мӢқмһ…лӢҲлӢӨ. CloudWatchк°Җ Metric Filter м—ҶмқҙлҸ„ мһҗлҸҷмңјлЎң metricsлҘј м¶”м¶ңн•©лӢҲлӢӨ. мҪ”л“ңлҘј мӮҙнҺҙліҙкІ мҠөлӢҲлӢӨ.

лӢӨмӢң logs нҒҙлқјмқҙм–ёнҠёлҘј мғқм„ұн•ҳкі  unix epochлЎң мӢңмҠӨн…ң мӢңк°„мқ„ к°Җм ёмҳөлӢҲлӢӨ.

//Create a reusable aws config that we can pass to our clients
let config = aws_config::load_defaults(BehaviorVersion::v2023_11_09()).await;

//Create a cloudwatch logs client
let client = cloudwatchlogs::Client::new(&config);

//get the time in unix epoch ms
let time_now = SystemTime::now()
    .duration_since(UNIX_EPOCH)
    .unwrap()
    .as_millis() as i64;

мқҙм ң EMF JSON л¬ёмһҗм—ҙмқ„ мғқм„ұн•©лӢҲлӢӨ. CloudWatchк°Җ м»ӨмҠӨн…Җ metricмқ„ мғқм„ұн•ҳлҠ” лҚ° н•„мҡ”н•ң лӘЁл“  лҚ°мқҙн„°лҘј нҸ¬н•Ён•ҙм•ј н•ҳлҜҖлЎң, namespace, dimensions, к°’мқ„ л¬ёмһҗм—ҙм—җ мһ„лІ л“ңн•©лӢҲлӢӨ.

//Create a json string in embedded metric format with our diceroll value.
let json_emf = json!(
    {
        "_aws": {
        "Timestamp": time_now,
        "CloudWatchMetrics": [
            {
            "Namespace": "rust_custom_metrics",
            "Dimensions": [["roll_value_emf_dimension"]],
            "Metrics": [
                {
                "Name": "roll_value_emf"
                }
            ]
            }
        ]
        },
        "roll_value_emf_dimension": roll_value.to_string(),
        "roll_value_emf": roll_value
    }
);

мЈјмӮ¬мң„ к°’мқ„ dimensionмңјлЎңлҸ„ мӮ¬мҡ©н•ҳкі  metric к°’мңјлЎңлҸ„ мӮ¬мҡ©н•ҳлҠ” м җм—җ мЈјлӘ©н•ҳм„ёмҡ”. мқҙл ҮкІҢ н•ҳл©ҙ мЈјмӮ¬мң„ к°’м—җ лҢҖн•ҙ GroupByлҘј мҲҳн–үн•ҳм—¬ к°Ғ к°’мқҙ лӘҮ лІҲ лӮҳмҷ”лҠ”м§Җ нҷ•мқён•  мҲҳ мһҲмҠөлӢҲлӢӨ.

мқҙм ң мқҙм „кіј к°ҷмқҖ л°©мӢқмңјлЎң APIлҘј нҳём¶ңн•ҳм—¬ log eventлҘј кё°лЎқн•©лӢҲлӢӨ:

let log_event = InputLogEvent::builder()
    .timestamp(time_now)
    .message(json_emf.to_string())
    .build();

let response = client
    .put_log_events()
    .log_group_name("rust_custom")
    .log_stream_name("diceroll_log_stream_emf")
    .log_events(log_event.unwrap())
    .send()
    .await?;

println!("EMF Log event submitted: {:?}", response);
Ok(())
